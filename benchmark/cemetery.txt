// device kernels
#define C1_WWG UINT64_C(0X5555555555555555)
#define C2_WWG UINT64_C(0x3333333333333333)
#define C3_WWG UINT64_C(0x0F0F0F0F0F0F0F0F)
#define C4_WWG UINT64_C(0x0101010101010101)
static inline uint64_t count_WWG_GPU(unsigned long long x) {
  x -= (x >> 1) & C1_WWG;
  x = ((x >> 2) & C2_WWG) + (x & C2_WWG);
  x = (x + (x >> 4)) & C3_WWG;
  x *= C4_WWG;

  return (x >> 56);
}

// Tree adder implementation found in https://metacpan.org/pod/Bit::Fast
#define C1_TRADD UINT64_C(0xAAAAAAAAAAAAAAAA)
#define C2_TRADD UINT64_C(0xCCCCCCCCCCCCCCCC)
#define C3_TRADD UINT64_C(0xF0F0F0F0F0F0F0F0)
#define C4_TRADD UINT64_C(0xFF00FF00FF00FF00)
#define C5_TRADD UINT64_C(0x00FF00FF00FF00FF)
#define C6_TRADD UINT64_C(0xFF00FF00FF00FF00)
#define C7_TRADD UINT64_C(0x0000FFFF0000FFFF)
#define C8_TRADD UINT64_C(0xFFFF0000FFFF0000)
#define C9_TRADD UINT64_C(0x00000000FFFFFFFF)
#define C10_TRADD UINT64_C(0xFFFFFFFF00000000)
static inline uint64_t tree_adder_GPU(unsigned long long v) {
  v = (v & C1_WWG) + ((v & C1_TRADD) >> 1);
  v = (v & C2_WWG) + ((v & C2_TRADD) >> 2);
  v = (v & C3_WWG) + ((v & C3_TRADD) >> 4);
  v = (v & C5_TRADD) + ((v & C6_TRADD) >> 8);
  v = (v & C7_TRADD) + ((v & C8_TRADD) >> 16);
  v = (v & C9_TRADD) + ((v & C10_TRADD) >> 32);
  return v;
}

#pragma omp declare target(tree_adder_GPU)

static inline uint64_t _inter_count_scalar_gpu(const uint64_t *restrict a,
                                               const uint64_t *restrict b,
                                               unsigned int n) {
  uint64_t cnt = 0;
  for (unsigned int i = 0; i < n; i++) {
#ifdef USE_BUILTIN_POPCOUNT
    cnt += (uint64_t)__builtin_popcountll(a[i] & b[i]);
#else
    cnt += (uint64_t)POPCOUNT(a[i] & b[i]);
#endif
  }
  return cnt;
}

/* -----------------------------------------------------------------------
 * Scalar Harley-Seal popcount for GPU target regions.
 *
 * Processes 16 x uint64_t words per iteration using carry-save adders
 * (CSA), accumulating bits into ones/twos/fours/eights/sixteens buckets.
 * Each bucket bit at level L represents L set bits in the input.
 *
 * Final count:
 *   total = 16*popcount(sixteens) + 8*popcount(eights)
 *         +  4*popcount(fours)    + 2*popcount(twos)
 *         +  1*popcount(ones)
 *
 * Applied here to the INTERSECTION (AND) of two bitset arrays a[] and b[],
 * so no separate AND pass is needed.
 *
 * Reference: Mula, Kurz, Lemire (2016) https://arxiv.org/abs/1611.07612
 * ----------------------------------------------------------------------- */

/* Scalar CSA: carry-save adder on uint64_t.
 * h = majority(a,b,c),  l = parity(a,b,c)  */
#define _HS_CSA(h, l, a, b, c)                                                 \
  do {                                                                         \
    uint64_t _u = (a) ^ (b);                                                   \
    (h) = ((a) & (b)) | (_u & (c));                                            \
    (l) = _u ^ (c);                                                            \
  } while (0)

#pragma omp declare target
static inline uint64_t
    harley_seal_inter_gpu(const uint64_t *restrict a,
                          const uint64_t *restrict b, unsigned int n) {
  uint64_t ones = 0, twos = 0, fours = 0;
  uint64_t eights = 0, sixteens = 0;
  uint64_t twosA, twosB, foursA, foursB, eightsA, eightsB;
  uint64_t cnt = 0;
  unsigned int i = 0;
  const unsigned int limit = n - (n % 16u);

  for (; i < limit; i += 16) {
    /* Load 16 AND-ed words */
    uint64_t w0 = a[i + 0] & b[i + 0];
    uint64_t w1 = a[i + 1] & b[i + 1];
    uint64_t w2 = a[i + 2] & b[i + 2];
    uint64_t w3 = a[i + 3] & b[i + 3];
    uint64_t w4 = a[i + 4] & b[i + 4];
    uint64_t w5 = a[i + 5] & b[i + 5];
    uint64_t w6 = a[i + 6] & b[i + 6];
    uint64_t w7 = a[i + 7] & b[i + 7];
    uint64_t w8 = a[i + 8] & b[i + 8];
    uint64_t w9 = a[i + 9] & b[i + 9];
    uint64_t w10 = a[i + 10] & b[i + 10];
    uint64_t w11 = a[i + 11] & b[i + 11];
    uint64_t w12 = a[i + 12] & b[i + 12];
    uint64_t w13 = a[i + 13] & b[i + 13];
    uint64_t w14 = a[i + 14] & b[i + 14];
    uint64_t w15 = a[i + 15] & b[i + 15];

    /* Harley-Seal CSA tree: 16 inputs → ones/twos/fours/eights/sixteens */
    _HS_CSA(twosA, ones, ones, w0, w1);
    _HS_CSA(twosB, ones, ones, w2, w3);
    _HS_CSA(foursA, twos, twos, twosA, twosB);
    _HS_CSA(twosA, ones, ones, w4, w5);
    _HS_CSA(twosB, ones, ones, w6, w7);
    _HS_CSA(foursB, twos, twos, twosA, twosB);
    _HS_CSA(eightsA, fours, fours, foursA, foursB);
    _HS_CSA(twosA, ones, ones, w8, w9);
    _HS_CSA(twosB, ones, ones, w10, w11);
    _HS_CSA(foursA, twos, twos, twosA, twosB);
    _HS_CSA(twosA, ones, ones, w12, w13);
    _HS_CSA(twosB, ones, ones, w14, w15);
    _HS_CSA(foursB, twos, twos, twosA, twosB);
    _HS_CSA(eightsB, fours, fours, foursA, foursB);
    _HS_CSA(sixteens, eights, eights, eightsA, eightsB);

    /* Drain sixteens bucket each iteration to prevent overflow */
    cnt += (uint64_t)__builtin_popcountll(sixteens);
    sixteens = 0;
  }

  /* Combine all bucket levels into final count */
  cnt = cnt * 16ULL + 8ULL * (uint64_t)__builtin_popcountll(eights) +
        4ULL * (uint64_t)__builtin_popcountll(fours) +
        2ULL * (uint64_t)__builtin_popcountll(twos) +
        (uint64_t)__builtin_popcountll(ones);

  /* Scalar tail: remaining < 16 words */
  for (; i < n; i++)
    cnt += (uint64_t)__builtin_popcountll(a[i] & b[i]);

  return cnt;
}

 *

#endif

#endif

                                  result, mode)                                \
  uint64_t count = 0;                                                          \
  _Alignas(ALIGNMENT) uint64_t setop_buffer[BUFFER_SIZE];                      \
  int l = 0;                                                                   \
                                                                               \
  if (mode == 0) { /* Mode 0: 32-bit Safe-Copy Path */                         \
    uint64_t safe_a[BUFFER_SIZE], safe_b[BUFFER_SIZE];                         \
    for (; l <= (int)(bit_size_in_qwords - BUFFER_SIZE); l += BUFFER_SIZE) {   \
      memcpy(safe_a, &a_row[l], BUFFER_SIZE * sizeof(uint64_t));               \
      memcpy(safe_b, &b_row[l], BUFFER_SIZE * sizeof(uint64_t));               \
      for (int k = 0; k < BUFFER_SIZE; k++) {                                  \
        setop_buffer[k] = safe_a[k] op safe_b[k];                              \
      }                                                                        \
      count += popcnt((void *)setop_buffer, BUFFER_SIZE * 8);                  \
    }                                                                          \
  } else if (mode == 1) { /* Mode 1: 64-bit Fully Aligned Path */              \
    for (; l <= (int)(bit_size_in_qwords - BUFFER_SIZE); l += BUFFER_SIZE) {   \
      _Pragma("omp simd aligned(a_row, b_row, setop_buffer: 64)") for (        \
          int k = 0; k < BUFFER_SIZE; k++) {                                   \
        setop_buffer[k] = a_row[l + k] op b_row[l + k];                        \
      }                                                                        \
      count += popcnt((void *)setop_buffer, BUFFER_SIZE * 8);                  \
    }                                                                          \
  } else { /* Mode 2: 64-bit Partially Aligned Path */                         \
    for (; l <= (int)(bit_size_in_qwords - BUFFER_SIZE); l += BUFFER_SIZE) {   \
      _Pragma("omp simd aligned(setop_buffer: 64)") for (int k = 0;            \
                                                         k < BUFFER_SIZE;      \
                                                         k++) {                \
        setop_buffer[k] = a_row[l + k] op b_row[l + k];                        \
      }                                                                        \
      count += popcnt((void *)setop_buffer, BUFFER_SIZE * 8);                  \
    }                                                                          \
  }                                                                            \
